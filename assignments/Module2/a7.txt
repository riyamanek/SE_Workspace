3.Create a pattern that prints a pyramid of stars (*) with 6 rows, centered like the loading animation you see on BookMyShow 
when a page is loading.<br><br><em><strong>Hint:</strong> Use spaces to align the stars in the center for each row.</em>

#include <stdio.h>

int main()
{
    for(int i=1;i<=6;i++)
    {
        for(int space=0;space<=6-i;space++)
        {
            printf(" ");
        }
        for(int j=1;j<=i;j++)
        {
            printf("* ");
       }
       printf("\n");
    }

    return 0;
}

(e.g., for a 4x4 grid, alternate 0 and 1 in each cell).

#include <stdio.h>

int main()
{
    for(int i=1;i<=4;i++)
    {
        for(int j=1;j<=4;j++)
        {

	        {

             }
