Create a constant variable to store the GST rate (for example, 18%) and use it to calculate the final price of a Zomato order with a given base price. The GST rate must not be changeable after its initial assignment.

#include <stdio.h>
#include <conio.h>
void main(){
    
     const float gst_rate=0.18;
     float baseprice=800;
     
     float gstamount=baseprice*gst_rate;
     float finalprice=gstamount+baseprice;
     
     printf("\nProduct Baseprice :- %f",baseprice);
     printf("\nProduct GSTAmount :- %f",gstamount);
     printf("\nProduct Finalprice :- %f",finalprice);
    getch();
}
