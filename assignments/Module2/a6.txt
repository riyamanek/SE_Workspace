1.Write a program that prints the numbers from 10 down to 1 using a for loop, similar to a countdown timer.

#include <stdio.h>
#include <conio.h>
int main()
{
    
    for(int i=10; i>=1; i--)
    {
        printf("\n %d", i);
    }
   return 0;
}
2.Create a menu-driven console app that lets the user: 1) View your favorite 3 IPL teams, 2) Add a new team, 3) Exit. Use a while loop to keep showing the menu until the user chooses Exit.<br><br><em><strong>Hint:</strong> Use input() (or Scanner in Java) to get the user's choice each time.</em>

#include <stdio.h>
#include <conio.h>
int main()
{
    int choice=0;
    char team[50];
    
    while(choice!=3)
    {
        printf("\n To show the Above Menu Driven");
        printf("\n Press 1 for View your favorite 3 IPL teams");
        printf("\n press 2 for Add a new team");
        printf("\n press 3 for Exit");
        
        printf("\n Enter Your Choice");
        scanf("%d",&choice);
        
        if(choice==1)
        {
            printf("\n favorite Ipl team");
            printf("\n 1 Mumbai Indians");
            printf("\n 2 Chennai Super King");
            printf("\n 3 Gujarat Titans ");
        }
        if(choice==2)
        {
            printf("\n Enter Your New IPL team");
            scanf("%s",&team);
            printf("\n Congrulations to add your new ipl team %s",team);
        }
        if(choice==3)
        {
            break;
        }
    
    }
    
   return 0;
}
3.Build a 'Guess the Song' game like Spotify — the program randomly picks a song name from a list and asks the user to guess it. Use a do-while loop so the user can keep guessing until they get it right.<br><br><em><strong>Constraint:</strong> Use at least 3 song names of your choice.</em>

#include <stdio.h>
#include <conio.h>
void main(){
    int choice;
    int guess=2;
    do{
        printf("\n Guess the song");
        printf("\n 1 for kesariya");
        printf("\n 2 for saiyara");
        printf("\n 3 for satranaga");
        printf("\n Enter your choice");
        scanf("%d",&choice);
        
        if(choice==guess)
        {
            printf("Correct");
        }
        else
        {
            printf("Wrong");
        }
    }
    while(choice!=guess);
}

Explain with your own example the difference between entry-controlled and exit-controlled loops by writing a short code snippet for each (for/while vs do-while) and describing what happens if the loop condition is false at the start.

Difference Between Entry-Controlled and Exit-Controlled Loops

Entry-Controlled Loop
The condition is checked before entering the loop.
for loop and while loop entry-controlled loop.

Exit-Controlled Loop
The condition is checked after executing the loop body.
do-while loop is an exit-controlled loop.


Entry-Controlled Loop:-
#include <stdio.h>
#include <conio.h>
int main(){
    int i=1;
    while(i<=5)
    {
        printf("\n %d",i);
        i++;
    }
    return 0;
}
